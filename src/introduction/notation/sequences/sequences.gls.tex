% Sequences glossary

\newglossaryentry{empty-seq}{
  name=\ensuremath{\eseq},
  description={the empty sequence}
}

\newglossaryentry{length-A}{
  name=\ensuremath{\len\seqA},
  description={the length of $\seqA$}
}

\newglossaryentry{concatenation-A-B}{
  name=\ensuremath{\seqA \conc \seqB},
  description={the concatenation of $\seqA$ and $\seqB$}
}

\newglossaryentry{sequence-of-a-b-c}{
  name=\ensuremath{\seq{\ea, \eb, \ec}},
  description={a sequence}
}

\newglossaryentry{without-A-B}{
  name=\ensuremath{\seqA \without \seqB},
  description={$\seqA$ without the elements of $\seqB$}
}
