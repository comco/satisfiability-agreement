% Functions glossary

\newglossaryentry{identity-A}{
  name=\ensuremath{\id\setA},
  description={the identity on $\setA$}
}

\newglossaryentry{total-function-f-A-B}{
  name=\ensuremath{\ff : \setA \to \setB},
  description={a total function from $\setA$ to $\setB$}
}

\newglossaryentry{injective-function-f-A-B}{
  name=\ensuremath{\ff : \setA \ito \setB},
  description={an injective function from $\setA$ into $\setB$}
}

\newglossaryentry{surjective-function-f-A-B}{
  name=\ensuremath{\ff : \setA \sto \setB},
  description={a surjective function from $\setA$ onto $\setB$}
}

\newglossaryentry{bijective-function-f-A-B}{
  name=\ensuremath{\ff : \setA \bto \setB},
  description={a bijective function between $\setA$ and $\setB$}
}

\newglossaryentry{partial-function-f-A-B}{
  name=\ensuremath{\ff : \setA \pto \setB},
  description={a partial function from $\setA$ to $\setB$}
}

\newglossaryentry{f-defined-a-b}{
  name=\ensuremath{\ff(\ea) \simeq \eb},
  description={$\ff$ is defined at $\ea$ with value $\eb$}
}

\newglossaryentry{f-undefined-a}{
  name=\ensuremath{\ff(\ea) \simeq \bot},
  description={$\ff$ is not defined at $\ea$}
}

\newglossaryentry{characteristic-function-S-A}{
  name=\ensuremath{\chfun\setA\setS},
  description={characteristic function}
}
