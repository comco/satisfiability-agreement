% Sets glossary

\newglossaryentry{cardinality-A}{
  name=\ensuremath{\card\setA},
  description={the cardinality of $\setA$}
}

\newglossaryentry{powerset-A}{
  name=\ensuremath{\pset\setA},
  description={the powerset of $\setA$}
}

\newglossaryentry{nepowerset-A}{
  name=\ensuremath{\nepset\setA},
  description={the set of nonempty subsets of $\setA$}
}

\newglossaryentry{k-powerset-A}{
  name=\ensuremath{\cpset\cardk\setA},
  description={the set of subsets of $\setA$ of cardinality $\cardk$}
}

\newglossaryentry{cartesian-product-A-B}{
  name=\ensuremath{\setA \cprod \setB},
  description={the cartesian product of $\setA$ and $\setB$}
}
