% Counters glossary

\newglossaryentry{counter-setup-C}{
  name=\ensuremath{\CS},
  description={a counter setup}
}

\newglossaryentry{data-C-A}{
  name=\ensuremath{\data\CS\StrA},
  description={$\CS$-data at $\StrA$}
}

\newglossaryentry{feqA-C-d-x}{
  name=\ensuremath{\feqA\CS\datad(\xx)},
  description={$\CS$-data at $\xx$ is $\datad$}
}

\newglossaryentry{feq-C-x-y}{
  name=\ensuremath{\feq\CS(\xx,\yy)},
  description={$\CS$-data equal at $\xx$ and $\yy$}
}

\newglossaryentry{fless-C-d-x}{
  name=\ensuremath{\fless\CS\datad(\xx)},
  description={$\CS$-data at $\xx$ less than $\datad$}
}

\newglossaryentry{fbetw-C-d-e-x}{
  name=\ensuremath{\fbetw\CS\datad\datae(\xx)},
  description={$\CS$-data at $\xx$ between $\datad$ and $\datae$}
}

\newglossaryentry{fallbetw-C-d-e}{
  name=\ensuremath{\fallbetw\CS\datad\datae},
  description={$\CS$-data between $\datad$ and $\datae$}
}

\newglossaryentry{fless-C-x-y}{
  name=\ensuremath{\fless\CS(\xx,\yy)},
  description={$\CS$-data at $\xx$ less than $\CS$-data at $\yy$}
}

\newglossaryentry{fsucc-C-x-y}{
  name=\ensuremath{\fsucc\CS(\xx,\yy)},
  description={$\CS$-data at $\yy$ succeeds $\CS$-data at $\xx$}
}
